SELECT Basics

The SELECT statement is the foundation of SQL queries. It retrieves data from one or more tables and is the most commonly used SQL command.

Code Examples

SQL
-- Select all columns from a table
SELECT * FROM employees;

-- Select all with table alias
SELECT * FROM employees e;

-- Select with schema prefix
SELECT * FROM hr.employees;

-- Limit results (different syntax per database)
-- MySQL/PostgreSQL:
SELECT * FROM employees LIMIT 10;

-- SQL Server:
SELECT TOP 10 * FROM employees;

-- Oracle:
SELECT * FROM employees WHERE ROWNUM <= 10;

-- Count all rows
SELECT COUNT(*) FROM employees;

-- Check if table has data
SELECT EXISTS (SELECT 1 FROM employees);
SQL
-- Select specific columns
SELECT first_name, last_name, salary
FROM employees;

-- Select with column order
SELECT last_name, first_name, email
FROM employees;

-- Select distinct values
SELECT DISTINCT department_id
FROM employees;

-- Select distinct combinations
SELECT DISTINCT department_id, job_id
FROM employees;

-- Count distinct values
SELECT COUNT(DISTINCT department_id)
FROM employees;

-- Select with table prefix (useful in joins)
SELECT employees.first_name, employees.last_name
FROM employees;

-- Using table alias
SELECT e.first_name, e.last_name
FROM employees e;
SQL
-- Column aliases
SELECT 
    first_name AS "First Name",
    last_name AS "Last Name",
    salary AS annual_salary
FROM employees;

-- Calculated columns
SELECT 
    first_name,
    last_name,
    salary,
    salary * 12 AS annual_salary,
    salary * 1.1 AS salary_with_raise
FROM employees;

-- String concatenation
-- Standard SQL:
SELECT first_name || ' ' || last_name AS full_name FROM employees;
-- MySQL:
SELECT CONCAT(first_name, ' ', last_name) AS full_name FROM employees;
-- SQL Server:
SELECT first_name + ' ' + last_name AS full_name FROM employees;

-- Literal values
SELECT 
    first_name,
    'Active' AS status,
    2024 AS year
FROM employees;

-- CASE expression in SELECT
SELECT 
    first_name,
    salary,
    CASE 
        WHEN salary > 10000 THEN 'High'
        WHEN salary > 5000 THEN 'Medium'
        ELSE 'Low'
    END AS salary_level
FROM employees;
